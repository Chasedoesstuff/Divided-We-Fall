# american colony asks a poorer neighbor the color of their bugatti
usunify.1 = {
	type = country_event
	placement = ROOT

	title = usunify.1.t
	desc = usunify.1.d
	flavor = usunify.1.f

	event_image = {
		video = "unspecific_signed_contract"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	duration = 3

	trigger = {
	}

	immediate = {
		random_neighbouring_state = {
			limit = {
				owner = {
					is_ai = yes
					relations:root >= relations_threshold:amicable
					OR = {
						country_has_primary_culture = cu:yankee
						country_has_primary_culture = cu:dixie
					}
					is_subject = yes
					NOT = { THIS = ROOT }
					top_overlord = {
						THIS = root.top_overlord
					}
					this.gdp < root.gdp
				}
			}
			owner = { save_scope_as = usa_annex_scope }
		}
		top_overlord = {
			save_scope_as = usa_overlord_country
			ruler = {
				save_scope_as = usa_overlord_ruler
			}
		}
	}

	option = {
		name = usunify.1.a
		default_option = yes
		annex = scope:usa_annex_scope
	}
}

		#switch to american flag icon

	trigger = {
	}

	immediate = {
			change_tag = USA
			if = {
			limit = {
				NOT = { country_has_primary_culture = cu:dixie }
