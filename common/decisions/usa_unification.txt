manifest_theft = {

	is_shown = {
		exists = c:USA
		root = c:USA
		is_subject = no
		NOT = { has_variable = stole_cuba }
		c:CUB = { owes_obligation_to = ROOT }
	}

	possible = {
		has_technology_researched = nationalism
		c:CUB = { owes_obligation_to = ROOT }
	}

	when_taken = {
	
		c:USA = {
			create_incident = {
				country = c:CUB
				value = 10
			}
		}
		
		c:USA = {
		create_diplomatic_pact = {
			country = c:CUB
			type = puppet
			} 
		create_diplomatic_pact = {
			country = c:PCO
			type = puppet
			}
							}
		c:CUB = {
			set_owes_obligation_to = {
				country = ROOT
				setting = no
			}
		}
		
		set_variable = stole_cuba
	}

	ai_chance = {
		base = 10
	}
}
